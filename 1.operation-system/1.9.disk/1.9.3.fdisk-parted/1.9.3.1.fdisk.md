[在线扩容磁盘]() 

~~~
[root@test ~]# df -h -T
[root@test ~]# fdisk -l
[root@test ~]# lsblk
[root@test ~]# yum install -y cloud-utils-growpart
[root@test ~]# growpart /dev/sdb 1                                                       // 在线扩容
[root@test ~]# lsblk
[root@test ~]# fdisk -l
[root@test ~]# xfs_growfs /dev/sdb1                                                     // resize2fs /dev/sdb1   (ext4系统文件类型所使用的扩展)
[root@test ~]# df -h -T 
~~~
