// enforcing(强制模式)，permissive(宽容模式)，disabled
[root@docker opt]# vi /etc/selinux/config
SELINUX=disabled
// 重启
[root@docker opt]# shutdown -r now
// 获取当前selinux状态
[root@docker opt]# getenforce
Enforcing
// 临时禁用
[root@docker opt]# setenforce 0
~~~
